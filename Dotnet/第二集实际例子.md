# .NET 8 后端项目与 RestSharp 自动化测试完整指南

## 1. 环境准备

### 安装必要的软件

1. 安装 .NET 8 SDK: https://dotnet.microsoft.com/download/dotnet/8.0
2. 安装 Visual Studio Code: https://code.visualstudio.com/
3. 在 VSCode 中安装以下扩展:
    - C# Dev Kit
    - .NET Core Test Explorer
    - REST Client (可选，用于手动测试API)


### 创建 API 控制器

创建 ApiProject/Controllers/TodosController.cs:

```csharp
using Microsoft.AspNetCore.Mvc;
using Microsoft.EntityFrameworkCore;
using ApiProject.Data;
using ApiProject.Models;

namespace ApiProject.Controllers
{
    [Route("api/[controller]")]
    [ApiController]
    public class TodosController : ControllerBase
    {
        private readonly AppDbContext _context;

        public TodosController(AppDbContext context)
        {
            _context = context;
        }

        // GET: api/Todos
        [HttpGet]
        public async Task<ActionResult<IEnumerable<Todo>>> GetTodos()
        {
            return await _context.Todos.ToListAsync();
        }

        // GET: api/Todos/5
        [HttpGet("{id}")]
        public async Task<ActionResult<Todo>> GetTodo(int id)
        {
            var todo = await _context.Todos.FindAsync(id);

            if (todo == null)
            {
                return NotFound();
            }

            return todo;
        }

        // PUT: api/Todos/5
        [HttpPut("{id}")]
        public async Task<IActionResult> PutTodo(int id, Todo todo)
        {
            if (id != todo.Id)
            {
                return BadRequest();
            }

            _context.Entry(todo).State = EntityState.Modified;

            try
            {
                await _context.SaveChangesAsync();
            }
            catch (DbUpdateConcurrencyException)
            {
                if (!TodoExists(id))
                {
                    return NotFound();
                }
                else
                {
                    throw;
                }
            }

            return NoContent();
        }

        // POST: api/Todos
        [HttpPost]
        public async Task<ActionResult<Todo>> PostTodo(Todo todo)
        {
            _context.Todos.Add(todo);
            await _context.SaveChangesAsync();

            return CreatedAtAction(nameof(GetTodo), new { id = todo.Id }, todo);
        }

        // DELETE: api/Todos/5
        [HttpDelete("{id}")]
        public async Task<IActionResult> DeleteTodo(int id)
        {
            var todo = await _context.Todos.FindAsync(id);
            if (todo == null)
            {
                return NotFound();
            }

            _context.Todos.Remove(todo);
            await _context.SaveChangesAsync();

            return NoContent();
        }

        private bool TodoExists(int id)
        {
            return _context.Todos.Any(e => e.Id == id);
        }
    }
}
```

### 创建测试基础设施

首先创建一个 WebApplicationFactory:

在 ApiProject.Tests 项目中创建 TestingWebAppFactory.cs 文件:

```csharp
using ApiProject.Data;
using Microsoft.AspNetCore.Hosting;
using Microsoft.AspNetCore.Mvc.Testing;
using Microsoft.EntityFrameworkCore;
using Microsoft.Extensions.DependencyInjection;

namespace ApiProject.Tests
{
    public class TestingWebAppFactory<TProgram> : WebApplicationFactory<TProgram> where TProgram : class
    {
        protected override void ConfigureWebHost(IWebHostBuilder builder)
        {
            builder.ConfigureServices(services =>
            {
                // 找到 DbContext 服务描述符
                var descriptor = services.SingleOrDefault(
                    d => d.ServiceType == typeof(DbContextOptions<AppDbContext>));

                if (descriptor != null)
                {
                    services.Remove(descriptor);
                }

                // 添加内存中数据库
                services.AddDbContext<AppDbContext>(options =>
                {
                    options.UseInMemoryDatabase("TestingDb");
                });

                // 构建服务提供者
                var sp = services.BuildServiceProvider();

                // 创建作用域
                using (var scope = sp.CreateScope())
                {
                    var scopedServices = scope.ServiceProvider;
                    var db = scopedServices.GetRequiredService<AppDbContext>();

                    // 确保数据库已创建
                    db.Database.EnsureCreated();

                    // 添加测试数据
                    if (!db.Todos.Any())
                    {
                        db.Todos.Add(new Models.Todo { Id = 1, Title = "Test Todo 1", IsComplete = false });
                        db.Todos.Add(new Models.Todo { Id = 2, Title = "Test Todo 2", IsComplete = true });
                        db.SaveChanges();
                    }
                }
            });
        }
    }
}
```

```csharp
                MaxTimeout = 10000
            };

        }
    }
}
```

### 创建 Todo 控制器测试

创建 IntegrationTests/TodosControllerTests.cs 文件:

```csharp
using System.Net;
using System.Threading.Tasks;
using ApiProject.Models;
using FluentAssertions;
using RestSharp;
using Xunit;

namespace ApiProject.Tests.IntegrationTests
{
    public class TodosControllerTests : BaseIntegrationTest
    {
        public TodosControllerTests(TestingWebAppFactory<Program> factory)
            : base(factory)
        {
        }

        [Fact]
        public async Task GetAllTodos_ReturnsSuccessStatusCode()
        {
            // 准备
            var request = new RestRequest("api/todos");

            // 执行
            var response = await RestClient.ExecuteGetAsync(request);

            // 断言
            response.StatusCode.Should().Be(HttpStatusCode.OK);
        }

        [Fact]
        public async Task GetTodoById_WhenTodoExists_ReturnsTodo()
        {
            // 准备
            var request = new RestRequest("api/todos/1");

            // 执行
            var response = await RestClient.ExecuteGetAsync<Todo>(request);

            // 断言
            response.Data.Should().NotBeNull();
            response.Data.Id.Should().Be(1);
            response.Data.Title.Should().Be("Test Todo 1");
        }

        [Fact]
        public async Task GetTodoById_WhenTodoDoesNotExist_ReturnsNotFound()
        {
            // 准备
            var request = new RestRequest("api/todos/999");

            // 执行
            var response = await RestClient.ExecuteGetAsync(request);

            // 断言
            response.StatusCode.Should().Be(HttpStatusCode.NotFound);
        }

        [Fact]
        public async Task CreateTodo_AddsNewTodo()
        {
            // 准备
            var request = new RestRequest("api/todos", Method.Post);
            var newTodo = new Todo
            {
                Title = "Test Create Todo",
                IsComplete = false
            };
            request.AddJsonBody(newTodo);

            // 执行
            var response = await RestClient.ExecutePostAsync<Todo>(request);

            // 断言
            response.StatusCode.Should().Be(HttpStatusCode.Created);
            response.Data.Should().NotBeNull();
            response.Data.Id.Should().BeGreaterThan(0);
            response.Data.Title.Should().Be(newTodo.Title);
        }

        [Fact]
        public async Task UpdateTodo_WhenTodoExists_UpdatesTodo()
        {
            // 准备
            // 先创建一个待办事项
            var createRequest = new RestRequest("api/todos", Method.Post);
            var newTodo = new Todo
            {
                Title = "Todo to be updated",
                IsComplete = false
            };
            createRequest.AddJsonBody(newTodo);
            var createResponse = await RestClient.ExecutePostAsync<Todo>(createRequest);
            var todoId = createResponse.Data.Id;

            // 更新待办事项
            var updateRequest = new RestRequest($"api/todos/{todoId}", Method.Put);
            var updatedTodo = new Todo
            {
                Id = todoId,
                Title = "Updated Todo",
                IsComplete = true
            };
            updateRequest.AddJsonBody(updatedTodo);

            // 执行
            var updateResponse = await RestClient.ExecutePutAsync(updateRequest);

            // 断言
            updateResponse.StatusCode.Should().Be(HttpStatusCode.NoContent);

            // 获取更新后的待办事项
            var getRequest = new RestRequest($"api/todos/{todoId}");
            var getResponse = await RestClient.ExecuteGetAsync<Todo>(getRequest);

            getResponse.Data.Should().NotBeNull();
            getResponse.Data.Title.Should().Be("Updated Todo");
            getResponse.Data.IsComplete.Should().BeTrue();
        }

        [Fact]
        public async Task DeleteTodo_WhenTodoExists_RemovesTodo()
        {
            // 准备
            // 先创建一个待办事项
            var createRequest = new RestRequest("api/todos", Method.Post);
            var newTodo = new Todo
            {
                Title = "Todo to be deleted",
                IsComplete = false
            };
            createRequest.AddJsonBody(newTodo);
            var createResponse = await RestClient.ExecutePostAsync<Todo>(createRequest);
            var todoId = createResponse.Data.Id;

            // 删除待办事项
            var deleteRequest = new RestRequest($"api/todos/{todoId}", Method.Delete);

            // 执行
            var deleteResponse = await RestClient.ExecuteAsync(deleteRequest);

            // 断言
            deleteResponse.StatusCode.Should().Be(HttpStatusCode.NoContent);

            // 验证待办事项已被删除
            var getRequest = new RestRequest($"api/todos/{todoId}");
            var getResponse = await RestClient.ExecuteGetAsync(getRequest);

            getResponse.StatusCode.Should().Be(HttpStatusCode.NotFound);
        }
    }
}
```

#### 添加认证测试

如果你的 API 使用认证，你可以在 RestSharp 请求中添加认证头:

```csharp
// 基本认证
request.AddHeader("Authorization", "Basic " + Convert.ToBase64String(Encoding.ASCII.GetBytes($"{username}:{password}")));

// Bearer Token 认证

```

```csharp
[Fact]

    // 断言

```

#### 参数化测试

```csharp
[Theory]
[InlineData(1, "Test Todo 1")]

{
    // 准备
    var request = new RestRequest($"api/todos/{id}");

    // 执行
```

#### 测试并发请求

```csharp
[Fact]
public async Task MakeConcurrentRequests_AllShouldSucceed()
{
    // 准备
    var tasks = new List<Task<RestResponse>>();
    for (int i = 0; i < 10; i++)
    {
        var request = new RestRequest("api/todos");
        tasks.Add(RestClient.ExecuteGetAsync(request));
    }

    // 执行
    await Task.WhenAll(tasks);

    // 断言
    foreach (var task in tasks)
    {
        var response = await task;
        response.StatusCode.Should().Be(HttpStatusCode.OK);
    }
}
```

```yaml
  push:

    branches: [ main ]

  build:
    runs-on: ubuntu-latest

    steps:
    - uses: actions/checkout@v3
    - name: Setup .NET 8
```

```csharp
using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Threading.Tasks;
using FluentAssertions;
using RestSharp;
using Xunit;
using Xunit.Abstractions;

namespace ApiProject.Tests.PerformanceTests
{
    public class TodosPerformanceTests : IntegrationTests.BaseIntegrationTest
    {
        public TodosPerformanceTests(TestingWebAppFactory<Program> factory, ITestOutputHelper output)
            : base(factory, output)
        {
        }

        [Fact]
        public async Task GetAllTodos_ShouldRespondUnder500ms()
        {
            // 准备
            var request = new RestRequest("api/todos");
            var stopwatch = new Stopwatch();

            // 执行
            stopwatch.Start();
            var response = await RestClient.ExecuteGetAsync(request);
            stopwatch.Stop();

            // 记录时间
            Output.WriteLine($"Request took: {stopwatch.ElapsedMilliseconds}ms");

            // 断言
            response.IsSuccessful.Should().BeTrue();
            stopwatch.ElapsedMilliseconds.Should().BeLessThan(500);
        }

        [Fact]
        public async Task GetTodos_Load_10RequestsUnder2Seconds()
        {
            // 准备
            var tasks = new List<Task>();
            var stopwatch = new Stopwatch();
            
            // 执行
            stopwatch.Start();
            for (int i = 0; i < 10; i++)
            {
                var request = new RestRequest("api/todos");
                tasks.Add(RestClient.ExecuteGetAsync(request));
            }
            await Task.WhenAll(tasks);
            stopwatch.Stop();
            
            // 记录时间
            Output.WriteLine($"10 requests took: {stopwatch.ElapsedMilliseconds}ms");
            
            // 断言
            stopwatch.ElapsedMilliseconds.Should().BeLessThan(2000);
        }
    }
}
```
